df_list <- df_list[names(df_list) %in% dhs7_mm.vec]

# child mortality
anyCMortalitySum.list <- lapply(df_list, function(x) {
  # df <- subset(x, everUnion ==0)
  cat_sum(x, "anyCMortality", useNA=F, roundby=2)
})


anyCMortality.sum <- summarySection(anyCMortalitySum.list, "anyCMortality::")
rm(anyCMortalitySum.list)

# Sibling mortality
anySMortalitySum.list <- lapply(df_list, function(x) {
  try(cat_sum(x, "anySMortality", useNA=F, roundby=2))
})

anySMortality.sum <- summarySection(anySMortalitySum.list, "anySMortality::")
rm(anySMortalitySum.list)

# sister death
sisterDeathSum.list <- lapply(df_list, function(x) {
  try(cat_sum(x, "sisterDeath", useNA=F, roundby=2))
})

sisterDeath.sum <- summarySection(sisterDeathSum.list, "sisterDeath::")
rm(sisterDeathSum.list)


table1_cat <- rbind(anyCMortality.sum, anySMortality.sum, sisterDeath.sum)
